```python
        monkeypatch.setattr(settings, "SMTP_PASSWORD", "pw")
        monkeypatch.setattr(settings, "SMTP_HOST", "smtp.example.com")
```

```python
    return SimpleNamespace(content=[block])

    "source_notes": "From zillow.com/preferred",
    "metrics": {
        "speed_to_action": {
            "label": "Speed to Action",
            "unit": "seconds",
            "target": 120,
            "yellow_floor": 180,
            "direction": "lower_is_better",

        target_file = tmp_path / "thresholds.json"

        target_file.write_text(json.dumps({"metrics": {"existing": "preserved"}}))
        original = target_file.read_text()

        monkeypatch.setattr(settings, "THRESHOLDS_FILE", target_file)

        # File unchanged
```

```python
"""Snapshot tests for renderers — guards against P3 refactor regressions."""

from pathlib import Path

import pytest


SNAPSHOT_DIR = Path(__file__).parent / "snapshots"


def _load_or_write(path: Path, actual: str, *, update: bool) -> str:
    """If snapshot file missing OR --update-snapshots, write actual; else return expected."""
    if update or not path.exists():
        path.parent.mkdir(parents=True, exist_ok=True)
        path.write_text(actual, encoding="utf-8")
        return actual
    return path.read_text(encoding="utf-8")


@pytest.fixture
def update_snapshots(request):
    return request.config.getoption("--update-snapshots", default=False)


def _fixed_agent():
    """Deterministic agent payload for snapshot stability."""
    return {
        "agent_id": "100",
        "name": "Jane Smith",
        "email": "jane@example.com",
        "period": "2026-04",
        "scored": {
            "csat": {"value": 0.85, "status": "green", "label": "CSAT"},
            "response_time": {"value": 120, "status": "yellow", "label": "Response Time"},
        },
        "overall": "yellow",
    }


class TestEmailSnapshot:
    def test_rendered_email_matches_snapshot(self, update_snapshots):
        from src.email_builder import build_email
        actual = build_email(_fixed_agent())
        snapshot = SNAPSHOT_DIR / "email_jane_2026-04.html"
        expected = _load_or_write(snapshot, actual, update=update_snapshots)
        assert actual == expected, (
            "Email render changed. Re-run with --update-snapshots if intentional."
        )


class TestDeckSnapshot:
    def test_rendered_deck_matches_snapshot(self, update_snapshots):
        from src.deck_builder import build_deck
        actual = build_deck([_fixed_agent()])
        snapshot = SNAPSHOT_DIR / "deck_2026-04.html"
        expected = _load_or_write(snapshot, actual, update=update_snapshots)
        assert actual == expected
```
